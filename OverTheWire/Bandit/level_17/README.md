```python
#!/usr/bin/env python

from pwn import *


def main():
    shell = ssh(host='bandit.labs.overthewire.org',
                user='bandit17',
                keyfile='bandit17.sshkey',
                port=2220
           )

    diff = shell['diff passwords.old passwords.new']
    
    print diff.split('\n')[-1].split()[1]

    shell.close()

if __name__ == "__main__":
    main()
```
